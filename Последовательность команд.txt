урок 2

урок 3
systemctl start mysqld

grep -i password /var/log/mysqld.log
mysql_secure_installation

systemctl enable php72-php-fpm
systemctl restart php72-php-fpm
systemctl enable nginx
systemctl restart nginx

урок 4
sudo nano /etc/my.cnf (редактирование master)
sudo systemctl restart mysqld
mysql -u root -p
	CREATE USER 'slave_user'@'192.168.1.2' IDENTIFIED BY 'Pass';
	alter user 'slave_u'@'192.168.1.2' IDENTIFIED WITH mysql_native_password by 'Pass';
	GRANT REPLICATION SLAVE ON *.* TO 'slave_u'@'192.168.1.2' IDENTIFIED BY 'Pass';
	FLUSH PRIVILEGES;

sudo nano /etc/my.cnf (редактирование slave)
sudo systemctl restart mysqld
mysql -u root -p
	CHANGE MASTER TO MASTER_HOST='192.168.1.1', MASTER_USER='slave_u', MASTER_PASSWORD='Pass', MASTER_LOG_FILE='mysql-bin.000002', MASTER_LOG_POS=1218;
	start slave;
